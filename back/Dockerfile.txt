# 베이스 이미지 지정
FROM python:3.10

# 컨테이너 내 작업 디렉토리 설정
WORKDIR /app/

# 의존성 파일 먼저 복사 및 설치 (캐시 활용)
COPY requirements.txt /app/
RUN pip install --no-cache-dir -r requirements.txt

# 나머지 모든 코드 복사
COPY . /app/

# FastAPI 포트 개방
EXPOSE 8000

# 컨테이너 실행 명령어
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
